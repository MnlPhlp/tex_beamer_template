% define title with section and subsection
\newcommand{\makeFrameTitle}{
    \setbox0=\hbox{\subsecname\unskip}\ifdim\wd0=0pt
        \frametitle{\secname}
    \else
        \frametitle{\secname}
        \framesubtitle{\subsecname}
    \fi
}
